⚙ Supabase Database Setup

Connect to Supabase so all users share the same data. Get your URL and anon key from supabase.com → Settings → API.

Setup: run this SQL in Supabase → SQL Editor:

Show SQL setup script 
create table if not exists drivers (id uuid primary key default gen_random_uuid(), name text not null, created_at timestamptz default now());
create table if not exists vehicles (id uuid primary key default gen_random_uuid(), truck_number text not null, trailer_number text not null, assigned_driver_id uuid, created_at timestamptz default now());
create table if not exists maintenance_records (id uuid primary key default gen_random_uuid(), vehicle_id uuid, service_date text, next_inspection_date text, notes text, created_at timestamptz default now());
create table if not exists brake_tests (id uuid primary key default gen_random_uuid(), vehicle_id uuid, test_date text, result text, notes text, created_at timestamptz default now());
create table if not exists tyre_records (id uuid primary key default gen_random_uuid(), vehicle_id uuid, photo_date text, readings jsonb, created_at timestamptz default now());
create table if not exists dot_inspections (id uuid primary key default gen_random_uuid(), vehicle_id uuid, driver_id uuid, inspection_date text, result text, notes text, created_at timestamptz default now());
create table if not exists mileage_records (id uuid primary key default gen_random_uuid(), vehicle_id uuid, driver_id uuid, mileage integer, date text, created_at timestamptz default now());
